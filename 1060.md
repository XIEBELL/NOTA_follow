```bash
sudo docker run -it --gpus all --shm-size=8g \
-v /home/xcb/AIC2024_Track1_Nota:/home/workspace/AIC2024_Track1_Nota \
-v /home/xcb/AIC2024_Track1_Nota/data:/workspace/ \
aic2024/track1_nota:latest /bin/bash
```
